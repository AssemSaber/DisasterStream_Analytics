## for specific model

docker compose exec airflow-scheduler dbt run --model model_name --project-dir /opt/airflow/dbt

ex)
# Run single model
docker compose exec airflow-scheduler dbt run --model stg_customers --project-dir /opt/airflow/dbt

# Run multiple models >> --model or --select 

docker compose exec airflow-scheduler dbt run --model stg_customers stg_orders --project-dir /opt/airflow/dbt

# Run all staging models
docker compose exec airflow-scheduler dbt run --model staging.* --project-dir /opt/airflow/dbt

# Run all marts models
docker compose exec airflow-scheduler dbt run --model marts.* --project-dir /opt/airflow/dbt

# Run all models with that tag, which maybe in different folders[staging,marts and so on]
docker compose exec airflow-scheduler dbt run --model tag:marketing --project-dir /opt/airflow/dbt
